\ifnum \Version=1
    \part If $A$ is $3\times 7$ and $Ax=b$ is consistent for all $b \in \mathbb R^3$, how many pivots does $A$ have? \framebox{\strut\hspace{1.0cm}}
    
    \ifnum \Solutions=1 {\color{DarkBlue} \textit{Solutions.} There must be a pivot in every row of $A$ so there are 3 pivots. 

    } 
   \else
   \fi
\fi

\ifnum \Version=2

    \part If $A$ is a $3\times7$ matrix, $b \in \mathbb R^3$ is in the span of the columns of $A$, $x \in \mathbb R^7$, and $A$ has exactly $2$ pivots, then how many free variables does $Ax=b$ have? \framebox{\strut\hspace{1.25cm}} 

    \ifnum \Solutions=1 {\color{DarkBlue} \textit{Solutions.} There are $7$ columns and $2$ pivots, so there must be 5 free variables. 
    } 
   \else
   \fi    

\fi

\ifnum \Version=3
    \part If $A$ is $5\times 7$ and $\vec x \to A\vec x$ is onto, how many pivots does $A$ have? \framebox{\strut\hspace{1.25cm}}

    \ifnum \Solutions=1 {\color{DarkBlue} \textit{Solutions.} There are $5$ rows and very row must be pivotal, so there must be 5 pivots. 
    } 
   \else
   \fi    
\fi

\ifnum \Version=4
    \part A is $12\times 7$ and $\vec x \to A\vec x$ is one-to-one. How many pivots does $A$ have? \framebox{\strut\hspace{1.25cm}}    
    
    \ifnum \Solutions=1 {\color{DarkBlue} \textit{Solutions.} Every column must be pivotal, so the answer is 7. 
    } 
   \else
   \fi        
\fi

\ifnum \Version=5
    \part Matrix $A$ is $12\times 7$ and $\vec x \to A\vec x$ is one-to-one. How many pivots does $A$ have? \framebox{\strut\hspace{1.25cm}}     
    
    \ifnum \Solutions=1 {\color{DarkBlue} \textit{Solutions.} 
    There are 7 columns and if the transform is one-to-one there must be a pivot in every column, so there are 7 pivots. 
    } 
   \else
      
   \fi    
\fi
