    \question[15] %2.5.11 verbatim 
    Consider the differential equation $\displaystyle \dydt = ky - y^3$, where $y$ is a real function of $t$, and $t \ge 0$, and $k>0$. 
    \begin{parts}
        \part State the critical points of the differential equation.  \vspace{0.75cm} 
        \part Draw the phase line, and determine whether the critical points (if any) are stable, semi-stable, or unstable. 
        \vspace{5cm} 
        \part Determine where $y$ is concave up and concave down for $y \ge 0$. 
        \vspace{6cm}
        \part Using results from parts (a) and (b), sketch several solution curves in the $ty$-plane for $y \ge 0$ and $t \ge 0$.
    \end{parts}

        \part[1] Express the system in the form $\displaystyle \frac{d\vec x}{dt} = A\vec x$, where $\vec x = \spalignmat{x;y}$. \vspace{2cm}
        \part[4] Determine the eigenvectors of $A$. 
        \vspace{5cm}
        \part[2] Express the general solution of the system $\frac{d\vec x}{dt} = A \vec x$ in terms of real valued functions. 
        \vspace{3cm} 
        \part[3] Sketch the phase portrait of the system. Please include the eigenspaces in your sketch and do not forget to label your axes.

    \end{parts}
