\question[5] An object is moving along a path specified by the curve $\mathbf r(t) = t^2 \mathbf i + t \mathbf j + (t^2+1) \mathbf k$ for $t \ge 0$. For each of the parts below please show your work. 

\begin{parts}
    \part Determine the velocity and acceleration of the object. 
    \ifnum \Solutions=1 {\color{DarkBlue} \\[12pt]
        Differentiating $\mathbf r(t)$ we obtain
        \begin{align}
          \mathbf v &= \mathbf r '(t) = 2t\mathbf i + \mathbf j + 2t\mathbf k \\
          \mathbf a &= \mathbf v ' (t) = 2\mathbf i+2\mathbf k
        \end{align}
        } 
    \else
      \vspace{4cm}
    \fi
    \part Determine the curvature of $\mathbf{r}(t)$ when $t=1$. You can use your results from part (a). 
    \ifnum \Solutions=1 {\color{DarkBlue} \\[12pt]
        Using the formula $\displaystyle \kappa = \frac{|\mathbf v \times \mathbf a|}{ |\mathbf v|^3 }$, then 
        \begin{align}
            \mathbf v \times \mathbf a &= \begin{vmatrix} i&j&k\\ 2t&1&2t \\2&0&2 \end{vmatrix} = \langle 2,0,-2 \rangle \\
            |\mathbf v \times \mathbf a |&= \sqrt{2^2+2^2} = \sqrt8 \\
            |\mathbf{v}| &= \sqrt{(2t)^2 + 1^2 + (2t)^2} = \sqrt{1+8t^2} \\
            \kappa &= \frac{|\mathbf v \times \mathbf a|}{ |\mathbf v|^3 }= \frac{\sqrt{8}}{ (1+8t^2)^{3/2}}
        \end{align}
        When $t=1$ we obtain $\displaystyle \kappa = \frac{\sqrt{8}}{(1+8)^{3/2}} = \frac{\sqrt{8}}{27}$.
        } 
    \else
      \vspace{4cm}
    \fi 
\end{parts}
\fi
