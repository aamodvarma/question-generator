% 12.3 and 12.4: Dot and Cross products

\ifnum \Version=1

\part The parallelogram determined by vectors $\mathbf a$ and $\mathbf b$ has area $\sqrt 2$. If $\mathbf a = \mathbf b+ \mathbf c$, and $\mathbf c$ makes an angle of $\frac{\pi}{4}$ with $\mathbf b$, and $|\mathbf c| = 6$, then $|\mathbf b | = $ \framebox{\strut\hspace{1cm}}. 

\ifnum \Solutions=1 {\color{DarkBlue} \textit{Answer:} $1/3$ \\[12pt] \textit{Solutions:} 
\begin{align*}
    \sqrt2&=|a\times b|\\
    &= |(b+c)\times b| \\
    &= |b\times b + c\times b| \\
    &= |c\times b|\\
    &= |c| \, |b| \sin\theta \\
    &= 6\, |b| \frac{1}{\sqrt2}\\
    \sqrt2\sqrt2&=6|b| \\
    |b| &= \frac26=\frac13
\end{align*}
}
\else
  
\fi
\fi







\ifnum \Version=2

\part The projection of $\mathbf u = \langle 2,1,2\rangle$ onto the vector $\mathbf v = \langle 1,0,1\rangle$ is $\text{proj}_{\mathbf v} \mathbf u = \langle a_1,a_2,a_3\rangle$, where $a_1 = \framebox{\strut\hspace{1cm}}$, $a_2 = \framebox{\strut\hspace{1cm}}$, $a_3 = \framebox{\strut\hspace{1cm}}$. A vector in the direction of $\mathbf u$ with length 18 is $\mathbf w = \langle c_1,c_2,c_3\rangle$ where $c_1 = \framebox{\strut\hspace{1cm}}$, $c_2 = \framebox{\strut\hspace{1cm}}$, $c_3 = \framebox{\strut\hspace{1cm}}$.

\ifnum \Solutions=1 {\color{DarkBlue} 
\textit{Solutions:} 
\begin{align*}
    \text{proj}_{\mathbf v} \mathbf u  
    &= \frac{\mathbf u \cdot \mathbf v}{\mathbf v \cdot \mathbf v}\mathbf v 
    = \frac{4}{2}\mathbf v = 2\mathbf v
\end{align*}
So $a_1 = 2$, $a_2=0$, $a_3=2$. And a unit vector in the direction of $\mathbf u$ is 
\begin{align}
    \hat u = \frac{1}{3} \begin{pmatrix} 2\\1\\2 \end{pmatrix}
\end{align}
So a vector with length 18 in the direction of $\mathbf u$ is 
\begin{align}
    18 \hat u 
    = \frac{18}{3} \begin{pmatrix} 2\\1\\2 \end{pmatrix} 
    = 6 \begin{pmatrix} 2\\1\\2 \end{pmatrix} 
    = \begin{pmatrix} 12\\6\\12 \end{pmatrix}
\end{align}
So $c_1 = 12$, $c_2=6$, $c_3=12$. 
}
\else
\fi
\fi

\ifnum \Version=3

\part Suppose $\mathbf u \times \mathbf v = \langle 2,3,4\rangle$ and $(2\mathbf u  + 3\mathbf v)\times \mathbf v = \mathbf w$. Then $\mathbf w = \langle c_1,c_2,c_3\rangle$ where $c_1 = \framebox{\strut\hspace{1cm}}$, $c_2 = \framebox{\strut\hspace{1cm}}$, $c_3 = \framebox{\strut\hspace{1cm}}$.

\ifnum \Solutions=1 {\color{DarkBlue} 
\textit{Solutions:} 
\begin{align*}
    \mathbf w 
    &= (2\mathbf u  + 3\mathbf v)\times \mathbf v \\ 
    &= 2\mathbf u \times \mathbf v + 3\mathbf v\times \mathbf v \\ 
    &= 2\mathbf u \times \mathbf v \\ 
    &= 2 \langle 2,3,4\rangle \\
    &= \langle 4,6,8\rangle
\end{align*}
So $c_1 = 4$, $c_2=6$, $c_3=8$. 
}
\else
\fi
\fi








\ifnum \Version=4

\part The parallelogram determined by vectors $\mathbf a$ and $\mathbf b$ has area $6$. If $\mathbf a = \mathbf b+ \mathbf c$, and $\mathbf c$ makes an angle of $\frac{\pi}{2}$ with $\mathbf b$, and $|\mathbf c| = 2$, then $|\mathbf b | = $ \framebox{\strut\hspace{1cm}}. 

\ifnum \Solutions=1 {\color{DarkBlue}   \textit{Solutions:} 
\begin{align*}
    6&=|a\times b|\\
    &= |(b+c)\times b| \\
    &= |b\times b + c\times b| \\
    &= |c\times b|\\
    &= |c| \, |b| \sin\theta \\
    &= 2\, |b| 1\\
    6&=2|b| \\
    |b| &= 3
\end{align*}
}
\else
  
\fi
\fi

\ifnum \Version=5

\part The cosine of the angle between the vectors $\mathbf a = \langle 2,1,2\rangle$ and $\mathbf b = \langle 1,0,2\rangle$ is $\cos \theta = \framebox{\strut\hspace{1cm}}$. 

\ifnum \Solutions=1 {\color{DarkBlue}  \textit{Solutions:} 
\begin{align*}
    \cos \theta &= \frac{\mathbf a \cdot \mathbf b }{|\mathbf a | \, |\mathbf b| } = \frac{2+4}{\sqrt{2^2+1^2+2^2} \ \sqrt{1^2+2^2}} = \frac{6}{\sqrt9 \sqrt{5}} = \frac{6}{3\sqrt5} = \frac{2}{\sqrt5}
\end{align*}
}
\else
  
\fi
\fi

\ifnum \Version=7
\part Given the points $P(2,0,1), Q(3,1,1), R(2,0,2)$, the area of triangle $PQR$ is \framebox{\strut\hspace{1.5cm}}. 

\ifnum \Solutions=1 {\color{DarkBlue} \textit{Answer:} $\sqrt2/2$. \\[12pt] \textit{Solutions:} 
$$area = \frac12 | PQ\times PR | = \frac12 \left| \begin{vmatrix} \mathbf i &\mathbf j &\mathbf k \\ 1&1&0\\0&0&1\end{vmatrix}\right| = \frac{\sqrt{1^2+(-1)^2+0}}{2} = \sqrt2/2$$
The area is $\sqrt2/2$. 
}

\fi

\ifnum \Version=8
\part The volume of the parallelepiped determined by $\langle 1,0,1\rangle, \langle 1,1,0\rangle, \langle 0,0,2 \rangle$ is \framebox{\strut\hspace{1.5cm}}. 

\ifnum \Solutions=1 {\color{DarkBlue}  \textit{Solutions:} Using the triple scalar product,
\begin{align*}
    V 
    &= |\langle 1,0,1\rangle \cdot (\langle 1,1,0\rangle \times \langle 0,0,2 \rangle ) |, \text{ and } \langle 1,1,0\rangle \times \langle 0,0,2 \rangle =  \begin{vmatrix}
        \textbf{i}& \textbf{j} & \textbf{k} \\ 1&1&0\\0&0&2\\\end{vmatrix} = \langle 2, -2,0\rangle\\
    V&= |\langle 1,0,1\rangle \cdot \langle 2, -2, 0\rangle  | \\
    &= |2| \\
    &= 2
\end{align*}
}

\fi
