\part The critical points of $f(x,y) = x^3-27x+y^2-4y+1$ are located at \framebox{\strut\hspace{4cm}}.

\ifnum \Solutions=1 {\color{DarkBlue} \textit{Answer:} $(\pm 3, 2)$ \\[12pt] \textit{Solutions.} $\nabla f = \langle 3x^2 - 27, 2y-4\rangle = \mathbf 0$. Thus,
\begin{align*}
    \langle 3x^2 - 27, 2y-4\rangle &= \langle 0,0\rangle \\
    \langle x^2 - 9, y-2\rangle &= \langle 0,0\rangle \\
    \langle x^2 , y\rangle &= \langle 9,2\rangle \\
    \langle x , y\rangle &= \langle \pm 3,2\rangle 
\end{align*}
The critical points are located at $(\pm 3, 2)$.

}

\fi
\fi

\ifnum \Version=5
% THOMAS 14.7
% FROM 2022
\part Classify the critical point of $f(x,y) = x^3-27x+y^2-6y+2$ at $(3,3)$. \framebox{\strut\hspace{3cm}}.

\ifnum \Solutions=1 {\color{DarkBlue} \textit{Answer:} Minimum. \\[12pt] \textit{Solutions.} 
\begin{align*}
    \nabla f &= \langle 3x^2 - 27, 2y-6\rangle \\
    f_{xx} & = 6x \\
    f_{yy} &= 2 \\
    f_{xy} &= 0 \\
    D (3,3)&= f_{xx}f_{yy} - f_{xy}^2 = 18\cdot 2 > 0 \ \Rightarrow \ \text{minimum}
\end{align*}
} 
\else
  
\fi
\fi

\ifnum \Version=7
% THOMAS 14.7
\part Let $f(x,y) = 3x^2+2y^3 - 6xy$. If possible, use the second derivative test to classify the critical point at the point $(1,1)$ \framebox{\strut\hspace{3cm}}, and at the point $(0,0)$ \framebox{\strut\hspace{3cm}}.

\ifnum \Solutions=1 {\color{DarkBlue} \textit{Solutions.} 
\begin{align*}
    \DFDX &=  6x - 6y  \quad \rightarrow \quad f_{xx}  = 6 \\
    \DFDY &= 6y^2 - 6x  \quad \rightarrow \quad f_{yy} = 12y \\
    f_{xy} &= -6 \\
    D(x,y) &= f_{xx}f_{yy} - f_{xy}^2 = 6\cdot12y - (-6)^2 = 72y - 36 
\end{align*}
    At the given points
\begin{align*} 
    D (1,1 )& > 0 , \quad  f_{xx} > 0 \ \Rightarrow \ \text{minimum} \\
    D (0,0)& <0 \ \Rightarrow \ \text{saddle} 
\end{align*}
} 
\else
  
\fi
\fi
